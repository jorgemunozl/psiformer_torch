\subsection{Schr\"odinger Equation}

%--------------------------------
% Frame 1.1.1: The Schr\"odinger equation
%--------------------------------
\begin{frame}{Why this exist}
\begin{itemize}
    \item With accurate solutions to the many body Schrodinger equation almost all the chemistry could be derived from first principles.
    \item Solve it analytically is impossible. The only path is do it numerically
    \item Challenging problem, standard method DFT, work with the density rather than the full function.
\end{itemize}
\end{frame}

\begin{frame}{The Schr\"odinger equation}
Let $\Psi$ be a complex value function called \textbf{wave function}, $\hat{H}$ the \textbf{Hamiltonian Operator}
and $E$ is the energy of the system, then:
\begin{equation}
\hat{H}\,\Psi(\mathbf{R})=E \Psi(\mathbf{R})
\end{equation}
Where $\mathbf{R}=\mathbf{r}\in\mathbb{R}^{3}$ is a system configuration.

\begin{block}{Hamiltonian Operator}
\begin{equation}
\hat{H}=\frac{\hat{\vec{{P}}}^{2}}{2m}+\hat{V} =-\frac{\hbar^{2}}{2m}\nabla^{2}+\hat{V}
\end{equation}
$\hat{V}$ is the potential energy operator, $\hat{\vec{P}}$ is the momentum operator and $\nabla$ is the laplacian operator.
\end{block}
\end{frame}

%--------------------------------
% Frame 1.1.2: Wave Function as Probability Density
%--------------------------------
\begin{frame}{Wave Function: Physical Meaning}

$\Psi(\mathbf{R},t)$ contains all available information about the quantum system.

\begin{block}{Observables}
An observable $O$ is represented by a Hermitian operator $\hat{O}$. The expectation value is:
\[
\langle O \rangle
=
\int \Psi^*(\mathbf{R},t)\,
\hat{O}\,
\Psi(\mathbf{R},t)\,
d^3R
\]
\end{block}

The most well known observables are \textbf{position, momentum, energy, angular momentum}.

\end{frame}

\end{figure}
\end{frame}

% Frame 1.1.3: Many-Body System
%--------------------------------
\begin{frame}{Many-Body System}
When considering $n$ bodies, we have:
\begin{equation}
\hat{H}\psi(\mathbf{x}_{1},\dots ,\mathbf{x}_{n})=E\psi(\mathbf{x}_{1},\dots ,\mathbf{x}_{n})
\end{equation}

With $\mathbf{x}_{i}=\{ \mathbf{r}_{i},\sigma \}$, where $\mathbf{r}_{i}\in \mathbb{R}^{3}$ is the position of each particle and $\sigma \in \{ \uparrow.\downarrow \}$ is the so called spin.

\begin{alertblock}{Considerations}
 \begin{itemize}
  \item Each particle interacts with all other particles.
  \item The solution must obey physical laws.
\end{itemize}
\end{alertblock}
\end{frame}

%--------------------------------
% Frame 1.1.4: Setting up the Hamiltonian
%--------------------------------
\begin{frame}{Setting up the Hamiltonian}
Hamiltonian for a molecule on atomic units:
\begin{align}
   \hat H   = & -\sum_{i=1}^{N}\frac{1}{2}\nabla_i^{2} +
-\sum_{I=1}^{N_n}\frac{1}{2M_I}\nabla_I^{2} \notag \\
& -\sum_{i=1}^{N}\sum_{I=1}^{N_n}\frac{Z_I}{|\mathbf r_i-\mathbf R_I|} \notag \\
& + \sum_{1\le i<j\le N}\frac{1}{|\mathbf r_i-\mathbf r_j|} \notag \\
& + \sum_{1\le I<J\le N_n}\frac{Z_I Z_J}{|\mathbf R_I-\mathbf R_J|} \notag
\end{align}
\end{frame}
\begin{frame}{Clever Approximations}
\begin{block}{Born Oppenheimer Approximation}
Assume nuclei are much heavier than electrons and treat nuclear positions as fixed parameters.
\begin{align}
\hat H_e
=-\sum_{i=0}^{N}\frac{1}{2}\nabla_i^{2}
-\sum_{i=0}^{N}\sum_{I=1}^{N_n}\frac{Z_I}{|\mathbf r_i-\mathbf R_I|}
+\sum_{0\le i<j\le N}\frac{1}{|\mathbf r_i-\mathbf r_j|}
+E_{nn}(\mathbf R)
\end{align}
\end{block}
Where:
\[
E_{nn}(\mathbf R)=\sum_{1\le I<J\le N_n}\frac{Z_I Z_J}{|\mathbf R_I-\mathbf R_J|}
\]
is constant for fixed nuclei.
\end{frame}

\subsection{Physical laws and conditions}
%--------------------------------
% Frame 1.1.6: Fermi-Dirac statistics
%--------------------------------
\begin{frame}{Fermi-Dirac statistics}
All fermions follow Fermi-Dirac statistics that implies the follow:
\begin{alertblock}{Antisymmetry requirement}
Exchanging two electrons must flip the sign of the wavefunction:
$$
\Psi(\dots,\mathbf{x}_i,\mathbf{x}_j,\dots)
=-\Psi(\dots,\mathbf{x}_j,\mathbf{x}_i,\dots)
$$
\end{alertblock}


\begin{block}{Slater Determinant}
One standard way to enforce antisymmetry is a determinant:
\begin{equation}
\Psi(\mathbf{x}_1,\dots,\mathbf{x}_n)=
\begin{vmatrix}
\phi_{1}^{k}(\mathbf{x}_{1})  & \dots  &  \phi_{1}^{k}(\mathbf{x}_{n}) \\
\vdots   &  & \vdots  \\
\phi_{n}^{k}(\mathbf{x}_{1}) & \dots & \phi_{n}^{k}(\mathbf{x}_{n})
\end{vmatrix}
\end{equation}
\end{block}
Where $\phi_{i}(\mathbf{x}_j)$ are called spin orbitals.
\end{frame}

%--------------------------------
% Frame 1.1.7: Kato cusp conditions
%--------------------------------
\begin{frame}{Kato cusp conditions}
The potential energy operator is:
$$
\hat{V}=-\sum_{i=1}^{N}\sum_{I=1}^{M}\frac{Z_I}{|\mathbf r_i-\mathbf R_I|}
+\sum_{1\le i<j\le N}\frac{1}{|\mathbf r_i-\mathbf r_j|}
$$
\begin{block}{Kato cusp constraints (atomic units)}
  Any wavefunction has a discontinuous derivative at point where particles coincide such as when electron meets nucleus or electron meets electrons.
\end{block}

Solution, introduce a \textbf{Jastrow factor} $\exp(\mathcal{J})$, $\mathcal{J}=u(r_{ij})$ finally.

$$
\Psi(R)=\exp(\mathcal{J})\det(\Phi)
$$

\end{frame}

\subsection{Optimizing an Ansatz}

%--------------------------------
% Frame 1.1.9: Loss: Variational Principle
%--------------------------------


\begin{frame}{How We Create Models in Deep Learning}
\textbf{Goal:} Learn parameters $\theta$ such that the model
$f_{\theta}(x)$ approximates the true underlying mapping
from inputs $x$ to targets $y$.
\begin{enumerate}
    \item Choose a model architecture
    \[
        f_{\theta}(x)
    \]
    with randomly initialized parameters $\theta$
    \item Define a loss function that measures prediction error. Need a dataset, $(x_i,y_i)$
    \[
        \mathcal{L}(\theta) = \frac{1}{N} \sum_{i=1}^{N}
        \ell\big(f_{\theta}(x_i), y_i\big)
    \]
    \item Compute gradients
    \[
        \nabla_{\theta} \mathcal{L}(\theta)
    \]
    \item Update parameters (gradient descent)
    \[
        \theta \leftarrow \theta - \eta \nabla_{\theta} \mathcal{L}(\theta)
    \]
    \item Repeat until convergence
\end{enumerate}
\end{frame}
\begin{frame}{Loss: Variational Principle}
Variational principle states:

$$E[\Psi] = \displaystyle \frac{\langle \Psi \mid H \mid \Psi \rangle}{\langle \Psi \mid \Psi \rangle} \ge E_0$$

Minimizing $E[\Psi]$ drives the ansatz toward the ground state.
$$
E[\Psi]=\mathcal{L}(\Psi_\theta)=\frac{\bra{\Psi_{\theta}} \hat{H}\ket{\Psi_{\theta}} }{\braket{ \Psi_{\theta}} }=\frac{\int d\mathbf{R}\Psi ^{*}(\mathbf{R})\hat{H}\Psi(\mathbf{R})}{\int d\mathbf{R}\Psi ^{*}(\mathbf{R})\Psi(\mathbf{R})}
$$
Define:
$$p_{\theta}(\mathbf{R})=|\Psi_{\theta}(\mathbf{R})|^{2}\frac{1}{\int  d\mathbf{R}'\Psi^{2}_{\theta}(\mathbf{R}')}\land E_{L}(\mathbf{R})=\frac{\hat{H}\Psi_{\theta}(\mathbf{R})}{\Psi_{\theta}(\mathbf{R})}
$$
Then:
\begin{equation}
\mathcal{L}_{\theta}=\mathbb{E}_{\mathbf{R}\sim |\Psi_{\theta}| ^{2}}[E_{L}(\mathbf{R})]
\end{equation}
\end{frame}
%--------------------------------
% Frame 1.1.10: Variational Monte Carlo
%--------------------------------
\begin{frame}{Variational Monte Carlo}
  \begin{block}{Quantum Monte Carlo}
With the samples $\mathbf{R}_{1},\dots,\mathbf{R}_{M}\sim |\Psi| ^{2}_{\theta}(\mathbf{R})$ we can make:

\begin{equation}
\mathcal{L}_{\theta}=\mathbb{E}_{\mathbf{R}\sim \Psi ^{2}_{\theta}}[E_{L}(\mathbf{R})]\approx \frac{1}{M}\sum_{i=1}^{M} E_{L}(\mathbf{R}_{i})
\end{equation}
\end{block}
With:
$$E_{L}(\mathbf{R}_{k})=\frac{\hat{H}\psi(\mathbf{R}_{k})}{\psi(\mathbf{R}_{k})}=-\frac{1}{2}\frac{\nabla^{2}\psi(\mathbf{R_{k}})}{\psi(\mathbf{R}_{k})}+V(\mathbf{R}_{k})
$$
\textbf{Obtain $\mathbf{R}_k$} $\to$ \textbf{Metropolis-Hastings Algorithm}
\end{frame}

%--------------------------------
% Frame 1.1.12: Metropolis-Hastings Algorithm
%--------------------------------
\begin{frame}
  \frametitle{Metropolis-Hastings Algorithm}
\textbf{Goal: Generate many samples} $\mathbf{R}\sim \rho$, Requirement: $C\rho$

1. $\mathbf{R}_{0}\in E$ arbitrary:
2. Propose $\mathbf{R}'=\mathbf{R}_{0}+\eta$ ,where $\eta \sim q(\eta)$, (Normal Gaussian)
3. Compute the quantity:
$$
A(\mathbf{R_{0}}, \mathbf{R}')=\text{min}\left( 1,\frac{\rho(\mathbf{R}')}{\rho(\mathbf{R}_{0})} \right)
$$
4. Generate a uniform number $U\in[0,1]$. If: $U<A(\mathbf{R}_{0} ,\mathbf{R'})$ then
$\mathbf{R_{1}}=\mathbf{R}'$, otherwise try another $\mathbf{R}'$. Accept or decline.

\end{frame}

\begin{frame}{Psiformer Ansatz}

\textbf{Proposed Model}

Motivated by these constraints, we propose a Slater--Jastrow form:
\[
\Psi_{\theta}(\mathbf{R}) =
\underbrace{\exp\big(\mathcal{J}_{\theta}(\mathbf{R})\big)}_{\text{Coulomb correlations}}
\,\times\,
\underbrace{\sum \omega_k  \det[ \boldsymbol{\phi}_{\theta}^{k}(\mathbf{R})]}_{\text{antisymmetry}}
\]
Using this specific form:

\begin{block}{Jastrow Factor}
\begin{equation}
 \mathcal{J}_{\theta}(x)=\sum_{i<j;\sigma_{i}=\sigma_{j}}-\frac{1}{4}\frac{\alpha^{2}_{par}}{\alpha_{par}+\lvert \mathbf{r}_{i}-\mathbf{r}_{j} \rvert }+\sum_{i,j;\sigma_{i}\neq \sigma_{j}}-\frac{1}{2}\frac{\alpha^{2}_{anti}}{\alpha_{anti}+\lvert \mathbf{r}_{i}-\mathbf{r}_{j} \rvert }
\end{equation}
\end{block}


\end{frame}

%--------------------------------
% Frame 2.3.1: Implementation : Laplacian Computation
%--------------------------------


\begin{frame}{Challenges Reproducing PsiFormer}

\begin{itemize}
    \item \textbf{Kinetic Energy Term}
    Computing the Laplacian
    \[
        \nabla^2 \Psi(\mathbf{R}) = \sum_i \frac{\partial^2 \Psi}{\partial R_i^2}
    \]
    is numerically unstable and computationally expensive.

    \vspace{0.2cm}

    \item \textbf{Determinant Stability}
    Slater determinant evaluation led to instability
    (overflow/underflow, ill-conditioned matrices).

    \vspace{0.2cm}

    \item \textbf{GPU Utilization}
    Find bottlenecks in order to use the 100\% of the GPU.
\end{itemize}

\end{frame}
